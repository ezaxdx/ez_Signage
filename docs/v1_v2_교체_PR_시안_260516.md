### 3. `lib/ai/v2/inputAdapter.ts` (신규) — v1 입력 → v2 변환
```ts
import { RecommendInput } from '@/lib/ai/recommendSignage'  // v1 입력 호환
import { RecommendInputV2 } from './recommendationLogic'

export function resolveLegacyToV2Input(input: RecommendInput): RecommendInputV2 {
  return {
    venue_key: slugify(input.venue),
    program_parts: input.programParts || [],
    event: {
      name: input.projectName,
      date: input.eventDate,
      expected_attendees: input.attendees,
      is_international: input.isInternational,
      has_vip: input.hasVip,
      session_count: input.sessionCount,
      shuttle_count: input.shuttleCount,
    },
    venue_specs: input.venueSpecs,
    floor_plan_analysis: input.floorPlanAnalysis,
  }
}
```
